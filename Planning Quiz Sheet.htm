        let currentQuestion = 0;

        let answers = new Array(questions.length).fill(null);

        let answered = new Array(questions.length).fill(false);

        let quizCompleted = false;

        function renderQuestion() {

            const content = document.getElementById('quiz-content');

            const q = questions[currentQuestion];

           

            let html = `

                <div class="question-container">

                    <div class="question-number">السؤال ${currentQuestion + 1} من ${questions.length}</div>

                    <div class="question-text">${q.question}</div>

                    <div class="options" id="options">

            `;

           

            q.options.forEach((option, index) => {

                const isSelected = answers[currentQuestion] === index;

                const isAnswered = answered[currentQuestion];

                const isCorrect = isAnswered && index === q.correct;

                const isIncorrect = isAnswered && isSelected && index !== q.correct;

               

                let optionClass = 'option';

                if (isSelected) optionClass += ' selected';

                if (isCorrect) optionClass += ' correct';

                if (isIncorrect) optionClass += ' incorrect';

               

                html += `

                    <div class="${optionClass}" onclick="selectOption(${index})">

                        <input type="radio" name="question" value="${index}" ${isSelected ? 'checked' : ''} ${isAnswered ? 'disabled' : ''}>

                        <label>${option}</label>

                    </div>

                `;

            });

           

            html += `

                    </div>

                    <div class="explanation ${answered[currentQuestion] ? 'show' : ''}" id="explanation">

                        <p><strong>الشرح:</strong> ${q.explanation}</p>

                    </div>

                </div>

            `;

           

            content.innerHTML = html;

           

            // تحديث شريط التقدم

            const progress = ((currentQuestion + 1) / questions.length) * 100;

            document.getElementById('progress').style.width = `${progress}%`;

           

            // تحديث أزرار التنقل

            document.getElementById('prevBtn').disabled = currentQuestion === 0;

           

            const nextBtn = document.getElementById('nextBtn');

            if (currentQuestion === questions.length - 1) {

                if (quizCompleted) {

                    nextBtn.textContent = 'عرض النتائج';

                } else {

                    nextBtn.textContent = 'إنهاء';

                }

            } else {

                nextBtn.textContent = 'التالي';

            }

        }

        function selectOption(optionIndex) {

            if (answered[currentQuestion] || quizCompleted) return;

           

            answers[currentQuestion] = optionIndex;

            answered[currentQuestion] = true;

           

            renderQuestion();

        }

        function nextQuestion() {

            if (currentQuestion < questions.length - 1) {

                currentQuestion++;

                renderQuestion();

            } else {

                // عرض النتائج

                showResults();

            }

        }

        function previousQuestion() {

            if (currentQuestion > 0) {

                currentQuestion--;

                renderQuestion();

            }

        }

        function showResults() {

            quizCompleted = true;

           

            // حساب النتيجة

            const correctAnswers = answers.reduce((count, answer, index) => {

                return count + (answer === questions[index].correct ? 1 : 0);

            }, 0);

           

            const percentage = (correctAnswers / questions.length) * 100;

           

            // إخفاء محتوى الاختبار

            document.getElementById('quiz-content').style.display = 'none';

            document.querySelector('.button-container').style.display = 'none';

           

            // عرض النتائج

            const resultsDiv = document.getElementById('results');

            resultsDiv.classList.add('show');

           

            document.getElementById('score').textContent = `${correctAnswers}/${questions.length}`;

           

            let feedbackText = '';

            if (percentage >= 90) {

                feedbackText = 'ممتاز! فهم عميق جداً لمفاهيم التخطيط والتفكير الاستراتيجي.';

            } else if (percentage >= 70) {

                feedbackText = 'جيد جداً! لديك فهم جيد للمفاهيم الأساسية.';

            } else if (percentage >= 50) {

                feedbackText = 'مقبول. تحتاج إلى مراجعة بعض المفاهيم.';

            } else {

                feedbackText = 'ننصح بمراجعة الدروس مرة أخرى للاستفادة بشكل أفضل.';

            }

           

            document.getElementById('feedback').innerHTML = `

                <p>${feedbackText}</p>

                <p style="margin-top: 10px; color: #666;">لقد أجبت بشكل صحيح على ${correctAnswers} من أصل ${questions.length} أسئلة.</p>

            `;

        }

        function restartQuiz() {

            currentQuestion = 0;

            answers = new Array(questions.length).fill(null);

            answered = new Array(questions.length).fill(false);

            quizCompleted = false;

           

            document.getElementById('quiz-content').style.display = 'block';

            document.querySelector('.button-container').style.display = 'flex';

            document.getElementById('results').classList.remove('show');

           

            renderQuestion();

        }
